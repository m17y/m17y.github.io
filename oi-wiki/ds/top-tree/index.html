  if (ans1)
    ans2 = x;
  else
    ans1 = x;
  return;
}

int qu[maxn], qv[maxn];

int main() {
  int i, TT = read(), n, I, U, V, x;
  long long ANS;
  for (I = 1; I <= TT; I++) {
    n = read();
    tot = n;
    ANS = 0;
    for (i = 1; i <= n; i++) ss[i] = 1;
    for (i = 1; i <= n - 1; i++) {
      qu[i] = U = read();
      qv[i] = V = read();
      lnk(U, V);
    }
    for (i = 1; i <= n - 1; i++) {
      cu(qu[i], qv[i]);
      ans1 = 0;
      ans2 = 0;
      non_local_search(qu[i], 0, 0, 0);
      ANS += ans1 + ans2;
      if (ans1) acs(ans1);
      if (ans2) acs(ans2);
      ans1 = 0;
      ans2 = 0;
      non_local_search(qv[i], 0, 0, 0);
      ANS += ans1 + ans2;
      if (ans1) acs(ans1);
      if (ans2) acs(ans2);
      lnk(qu[i], qv[i]);
    }
    cout << ANS << '\n';
    for (i = 1; i <= tot; i++)
      T[i][0] = T[i][1] = T[i][2] = ss[i] = r[i] = maxs[i] = f[i] = 0;
    tot = top = 0;
  }
  return 0;
}
